48  {
49  string_view_t payload;
50  memory_buf_t formatted;
51  if (enable_formatting_)
52  {
53  base_sink<Mutex>::formatter_->format(msg, formatted);
54  payload = string_view_t(formatted.data(), formatted.size());
55  }
56  else
57  {

59  }
60

62  // limit to max int
63  if (length > static_cast<size_t>(std::numeric_limits<int>::max()))
64  {
65  length = static_cast<size_t>(std::numeric_limits<int>::max());
66  }
67 
68  ::syslog(syslog_prio_from_level(msg), "%.*s", static_cast<int>(length), payload.data());
69  }
