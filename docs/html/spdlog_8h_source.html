27 // Create and register a logger with a templated sink type
28 // The logger's level, formatter and flush level will be set according the
29 // global settings.
30 //
31 // Example:
32 // spdlog::create<daily_file_sink_st>("logger_name", "dailylog_filename", 11, 59);
33 template<typename Sink, typename... SinkArgs>
34 inline std::shared_ptr<spdlog::logger> create(std::string logger_name, SinkArgs &&... sink_args)
35 {
36  return default_factory::create<Sink>(std::move(logger_name), std::forward<SinkArgs>(sink_args)...);
37 }

105 // API for using default logger (stdout_color_mt),
106 // e.g: spdlog::info("Message {}", 1);
107 //
108 // The default logger object can be accessed using the spdlog::default_logger():
109 // For example, to add another sink to it:
110 // spdlog::default_logger()->sinks()->push_back(some_sink);
111 //
112 // The default logger can replaced using spdlog::set_default_logger(new_logger).
113 // For example, to replace it with a file logger.
114 //
115 // IMPORTANT:
116 // The default API is thread safe (for _mt loggers), but:
117 // set_default_logger() *should not* be used concurrently with the default API.
118 // e.g do not call set_default_logger() from one thread while calling spdlog::info() from another.
