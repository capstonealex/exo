177 SPDLOG_INLINE void logger::sink_it_(const details::log_msg &msg)
178 {
179  for (auto &sink : sinks_)
180  {
181  if (sink->should_log(msg.level))
182  {
183  SPDLOG_TRY
184  {
185  sink->log(msg);
186  }
187  SPDLOG_LOGGER_CATCH()

189  }

194  }

226 SPDLOG_INLINE void logger::err_handler_(const std::string &msg)
227 {
228  if (custom_err_handler_)
229  {
230  custom_err_handler_(msg);
231  }
232  else
233  {
234  using std::chrono::system_clock;
235  static std::mutex mutex;
236  static std::chrono::system_clock::time_point last_report_time;
237  static size_t err_counter = 0;
238  std::lock_guard<std::mutex> lk{mutex};
239  auto now = system_clock::now();
240  err_counter++;
241  if (now - last_report_time < std::chrono::seconds(1))
242  {
243  return;
244  }
245  last_report_time = now;
246  auto tm_time = details::os::localtime(system_clock::to_time_t(now));
247  char date_buf[64];
248  std::strftime(date_buf, sizeof(date_buf), "%Y-%m-%d %H:%M:%S", &tm_time);
249  fprintf(stderr, "[*** LOG ERROR #%04zu ***] [%s] [%s] {%s}\n", err_counter, date_buf, name().c_str(), msg.c_str());
250  }
