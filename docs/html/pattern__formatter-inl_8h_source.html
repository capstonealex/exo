35 class scoped_padder
36 {
37 public:
38  scoped_padder(size_t wrapped_size, const padding_info &padinfo, memory_buf_t &dest)
39  : padinfo_(padinfo)
40  , dest_(dest)
41  {
42  remaining_pad_ = static_cast<long>(padinfo.width_) - static_cast<long>(wrapped_size);
43  if (remaining_pad_ <= 0)
44  {
45  return;
46  }
47 
48  if (padinfo_.side_ == padding_info::left)
49  {
50  pad_it(remaining_pad_);
51  remaining_pad_ = 0;
52  }
53  else if (padinfo_.side_ == padding_info::center)
54  {
55  auto half_pad = remaining_pad_ / 2;
56  auto reminder = remaining_pad_ & 1;
57  pad_it(half_pad);
58  remaining_pad_ = half_pad + reminder; // for the right side
59  }
60  }
61 
62  ~scoped_padder()
63  {
64  if (remaining_pad_ >= 0)
65  {
66  pad_it(remaining_pad_);
67  }
68  else if (padinfo_.truncate_)
69  {
70  long new_size = static_cast<long>(dest_.size()) + remaining_pad_;
71  dest_.resize(static_cast<size_t>(new_size));
72  }
73  }
74 
75 private:
76  void pad_it(long count)
77  {
78  fmt_helper::append_string_view(string_view_t(spaces_.data(), static_cast<size_t>(count)), dest_);
79  }
80 
81  const padding_info &padinfo_;
82  memory_buf_t &dest_;
83  long remaining_pad_;
84  string_view_t spaces_{" ", 64};
85 };

579 // ISO 8601 offset from UTC in timezone (+-HH:MM)
580 template<typename ScopedPadder>
581 class z_formatter final : public flag_formatter
582 {
583 public:
584  explicit z_formatter(padding_info padinfo)
585  : flag_formatter(padinfo)

592  void format(const details::log_msg &msg, const std::tm &tm_time, memory_buf_t &dest) override
593  {
594  const size_t field_size = 6;
595  ScopedPadder p(field_size, padinfo_, dest);
596 
597  auto total_minutes = get_cached_offset(msg, tm_time);
598  bool is_negative = total_minutes < 0;
599  if (is_negative)
600  {
601  total_minutes = -total_minutes;
602  dest.push_back('-');
603  }
604  else
605  {
606  dest.push_back('+');
607  }
608 
609  fmt_helper::pad2(total_minutes / 60, dest); // hours
610  dest.push_back(':');
611  fmt_helper::pad2(total_minutes % 60, dest); // minutes
612  }
613 
614 private:
615  log_clock::time_point last_update_{std::chrono::seconds(0)};
616  int offset_minutes_{0};
617 
618  int get_cached_offset(const log_msg &msg, const std::tm &tm_time)
619  {
620  // refresh every 10 seconds
621  if (msg.time - last_update_ >= std::chrono::seconds(10))
622  {
623  offset_minutes_ = os::utc_minutes_offset(tm_time);
624  last_update_ = msg.time;
625  }
626  return offset_minutes_;
627  }
