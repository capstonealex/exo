43 inline void pad2(int n, memory_buf_t &dest)
44 {
45  if (n > 99)
46  {
47  append_int(n, dest);
48  }
49  else if (n > 9) // 10-99
50  {
51  dest.push_back(static_cast<char>('0' + n / 10));
52  dest.push_back(static_cast<char>('0' + n % 10));
53  }
54  else if (n >= 0) // 0-9
55  {
56  dest.push_back('0');
57  dest.push_back(static_cast<char>('0' + n));
58  }
59  else // negatives (unlikely, but just in case, let fmt deal with it)
60  {
61  fmt::format_to(dest, "{:02}", n);
62  }
63 }

65 template<typename T>
66 inline void pad_uint(T n, unsigned int width, memory_buf_t &dest)
67 {
68  static_assert(std::is_unsigned<T>::value, "pad_uint must get unsigned T");
69  auto digits = count_digits(n);
70  if (width > digits)
71  {
72  const char *zeroes = "0000000000000000000";
73  dest.append(zeroes, zeroes + width - digits);
